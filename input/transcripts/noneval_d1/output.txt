from __future__ import annotations

import argparse
from pathlib import Path

import torch

from c4az.game import Position
from c4az.mcts import MCTSConfig, run_mcts
from c4az.net import Connect4Net


def load_model(checkpoint_path: Path, device: torch.device) -> Connect4Net:
    checkpoint = torch.load(checkpoint_path, map_location=device)
    architecture = checkpoint.get("architecture", {})
    model = Connect4Net(
        channels=int(architecture.get("channels", 96)),
        blocks=int(architecture.get("blocks", 6)),
    ).to(device)
    model.load_state_dict(checkpoint["model_state"])
    model.eval()
    return model


def parse_args() -> argparse.Namespace:
    parser = argparse.ArgumentParser(description="Play Connect 4 against the trained model.")
    parser.add_argument("--checkpoint", type=Path, default=Path("/runs/long_run/best.pt"))
    parser.add_argument("--simulations", type=int, default=768)
    parser.add_argument("--human-first", action="store_true")
    parser.add_argument("--model-first", action="store_true")
    return parser.parse_args()


def main() -> None:
    args = parse_args()
    if args.human_first and args.model_first:
        raise SystemExit("Choose only one of --human-first or --model-first.")

    human_first = args.human_first
    if not args.human_first and not args.model_first:
        human_first = False

    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    model = load_model(args.checkpoint, device)
    config = MCTSConfig(simulations=args.simulations, temperature=0.0, dirichlet_epsilon=0.0)
    position = Position()

    print("Connect 4")
    print("Type a column number from 1 to 7.")
    print("The board is shown from the physical front view.")
    print()

    while not position.is_terminal():
        print(position.render(player_one_token="R", player_two_token="Y"))
        player_one_turn = position.moves % 2 == 0
        human_turn = player_one_turn if human_first else not player_one_turn

        if human_turn:
            while True:
                raw = input("Your move (1-7): ").strip()
                if raw not in {"1", "2", "3", "4", "5", "6", "7"}:
                    print("Enter a number from 1 to 7.")
                    continue
                action = int(raw) - 1
                if not position.can_play(action):
                    print("That column is full. Pick another one.")
                    continue
                break
        else:
            result = run_mcts(position, model, device, config)
            action = result.action
            print(f"Model plays column {action + 1}.")

        position = position.play(action)
        print()

    print(position.render(player_one_token="R", player_two_token="Y"))
    outcome = position.outcome()
    if outcome == 0:
        print("Draw.")
        return

    winner = position.winner()
    human_player = 1 if human_first else 2
    human_won = winner == human_player
    print("You win." if human_won else "Model wins.")


if __name__ == "__main__":
    main()
