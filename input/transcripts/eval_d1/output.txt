import json
from pathlib import Path
import numpy as np
import torch
from tqdm import tqdm
from connect4az.board import Connect4State, state_from_moves
from connect4az.data import load_dataset
from connect4az.mcts import MCTSConfig, ModelEvaluator, run_mcts
from connect4az.model_io import load_checkpoint, save_checkpoint
from connect4az.oracle import SolverOracle
run_dir = Path('/workspace/runs/final_long_run')
checkpoint = run_dir / 'refine_best.pt'
device = 'cuda' if torch.cuda.is_available() else 'cpu'
model, _ = load_checkpoint(checkpoint, device=device)
evaluator = ModelEvaluator(model, device=device)
oracle = SolverOracle()
_, _, _, moves = load_dataset(run_dir / 'bootstrap_dataset.npz')
rng = np.random.default_rng(209)
indices = rng.choice(len(moves), size=256, replace=False)
states = [state_from_moves(str(moves[idx])) for idx in indices]
search_config = MCTSConfig(num_simulations=256, c_puct=1.5, dirichlet_alpha=1.1,
dirichlet_eps=0.0)
optimal_moves = 0
preserved_non_loss = 0
root_winning_positions = 0
total_regret = 0.0
total_positions = 0
for state in tqdm(states, desc='final-eval', leave=False):
    best_score, scores = oracle.analyze(state.moves)
    if best_score < -900:
        continue
    policy, _ = run_mcts(state, evaluator, search_config, add_noise=False)
    action = int(np.argmax(policy))
    chosen_score = int(scores[action])
    optimal_moves += int(chosen_score == best_score)
    if best_score >= 0:
        root_winning_positions += 1
        preserved_non_loss += int(chosen_score >= 0)
    total_regret += float(best_score - chosen_score)
    total_positions += 1
opening_config = MCTSConfig(num_simulations=1024, c_puct=1.5,
dirichlet_alpha=1.1, dirichlet_eps=0.0)
root = Connect4State()
root_policy, _ = run_mcts(root, evaluator, opening_config, add_noise=False)
reply_actions = {}
for human in range(7):
    state = root.play(human)
    policy, _ = run_mcts(state, evaluator, opening_config, add_noise=False)
    reply_actions[str(human + 1)] = int(np.argmax(policy)) + 1
metrics = {
    'positions': float(total_positions),
    'optimal_move_rate': optimal_moves / max(1, total_positions),
    'non_losing_preservation_rate': preserved_non_loss / max(1,
root_winning_positions),
    'average_regret': total_regret / max(1, total_positions),
    'opening_first_move': int(np.argmax(root_policy)) + 1,
    'opening_first_move_policy': [float(x) for x in root_policy.tolist()],
    'opening_reply_actions_at_1024_sims': reply_actions,
    'evaluation_note': 'Opening exact-score metrics omitted here because full
solver checks on the root opening tree exceeded practical memory/time limits in
this environment.',
}
(run_dir / 'final_metrics.json').write_text(json.dumps(metrics, indent=2,
sort_keys=True), encoding='utf-8')
save_checkpoint(run_dir / 'final_model.pt', model, None, 0, metrics)
print(json.dumps(metrics, indent=2, sort_keys=True))
