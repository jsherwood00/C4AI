    153 +    autocast_ctx = torch.autocast(device_type="cuda", dtype=torch.bfloat16) if device.type == "cuda" else nullcontext()
    154 +    for planes, policies, values in loader:
    155 +        planes = planes.to(device)
    156 +        policies = policies.to(device)
    157 +        values = values.to(device)
    158 +        optimizer.zero_grad(set_to_none=True)
    159 +        with autocast_ctx:
    160 +            logits, predicted_values = model(planes)
    161 +            policy_loss = soft_cross_entropy(logits, policies)
    162 +            value_loss = torch.mean((predicted_values - values) ** 2)
    163 +            loss = policy_loss + value_weight * value_loss
    164 +        loss.backward()
    165 +        clip_grad_norm_(model.parameters(), max_norm=1.0)
    166 +        optimizer.step()
    167 +
    168 +        total_loss += float(loss.item())
    169 +        total_policy_loss += float(policy_loss.item())
    170 +        total_value_loss += float(value_loss.item())
    171 +        picks = torch.argmax(logits, dim=-1)
    172 +        total_optimal += float((policies[torch.arange(policies.size(0)), picks] > 0).float().mean().item())
    173 +        batches += 1
    174 +    return {
    175 +        "loss": total_loss / max(1, batches),
    176 +        "policy_loss": total_policy_loss / max(1, batches),
    177 +        "value_loss": total_value_loss / max(1, batches),
    178 +        "optimal_top1": total_optimal / max(1, batches),
    179 +    }

    182 +@torch.no_grad()
    183 +def evaluate_epoch(
    184 +    model: nn.Module,
    185 +    loader: DataLoader[tuple[Tensor, Tensor, Tensor]],
    186 +    device: torch.device,
    187 +    value_weight: float,
    188 +) -> dict[str, float]:
    189 +    model.eval()
    190 +    total_loss = 0.0
    191 +    total_policy_loss = 0.0
    192 +    total_value_loss = 0.0
    193 +    total_optimal = 0.0
    194 +    total_value_sign = 0.0
    195 +    batches = 0
    196 +    autocast_ctx = torch.autocast(device_type="cuda", dtype=torch.bfloat16) if device.type == "cuda" else nullcontext()
    197 +    for planes, policies, values in loader:
    198 +        planes = planes.to(device)
    199 +        policies = policies.to(device)
    200 +        values = values.to(device)
    201 +        with autocast_ctx:
    202 +            logits, predicted_values = model(planes)
    203 +            policy_loss = soft_cross_entropy(logits, policies)
    204 +            value_loss = torch.mean((predicted_values - values) ** 2)
    205 +            loss = policy_loss + value_weight * value_loss
    206 +        total_loss += float(loss.item())
    207 +        total_policy_loss += float(policy_loss.item())
    208 +        total_value_loss += float(value_loss.item())
    209 +        picks = torch.argmax(logits, dim=-1)
    210 +        total_optimal += float((policies[torch.arange(policies.size(0)), picks] > 0).float().mean().item())
    211 +        predicted_sign = torch.where(predicted_values > 0.15, 1, torch.where(predicted_values < -0.15, -1, 0))
    212 +        target_sign = torch.where(values > 0.5, 1, torch.where(values < -0.5, -1, 0))
    213 +        total_value_sign += float((predicted_sign == target_sign).float().mean().item())
    214 +        batches += 1
    215 +    return {
    216 +        "loss": total_loss / max(1, batches),
    217 +        "policy_loss": total_policy_loss / max(1, batches),
    218 +        "value_loss": total_value_loss / max(1, batches),
    219 +        "optimal_top1": total_optimal / max(1, batches),
    220 +        "value_sign_acc": total_value_sign / max(1, batches),
    221 +    }

    223 +
    224 +@torch.no_grad()
    225 +def evaluate_mcts(
    226 +    model: nn.Module,
    227 +    benchmark: list[LabeledPosition],
    228 +    device: torch.device,
    229 +    num_simulations: int,
    230 +) -> dict[str, float]:
    231 +    mcts = MCTS(model=model, device=device, num_simulations=num_simulations, c_puct=1.5)
    232 +    optimal = 0
    233 +    value_match = 0
    234 +    average_value_error = 0.0
    235 +    for record in benchmark:
    236 +        state = record.state()
    237 +        move, _, value = mcts.select_move(state, temperature=0.0)
    238 +        if record.policy[move] > 0:
    239 +            optimal += 1
    240 +        target_sign = 0 if abs(record.value) < 0.5 else (1 if record.value > 0 else -1)
    241 +        predicted_sign = 0 if abs(value) < 0.2 else (1 if value > 0 else -1)
    242 +        if target_sign == predicted_sign:
    243 +            value_match += 1
    244 +        average_value_error += abs(float(record.value) - float(value))
    245 +    count = max(1, len(benchmark))
    246 +    return {
    247 +        "optimal_rate": optimal / count,
    248 +        "value_match_rate": value_match / count,
    249 +        "mean_value_error": average_value_error / count,
    250 +    }

    253 +def generate_self_play_examples(
    254 +    model: nn.Module,
    255 +    device: torch.device,
    256 +    num_games: int,
    257 +    num_simulations: int,
    258 +    seed: int,
    259 +    temperature_moves: int,
    260 +) -> tuple[np.ndarray, np.ndarray, np.ndarray, dict[str, float]]:
    261 +    rng = np.random.default_rng(seed)
    262 +    all_planes: list[np.ndarray] = []
    263 +    all_policies: list[np.ndarray] = []
    264 +    all_values: list[float] = []
    265 +    lengths: list[int] = []
    266 +    winners: list[int] = []
    267 +
    268 +    for _ in tqdm(range(num_games), desc="self-play", leave=False):
    269 +        state = Connect4State.initial()
    270 +        history: list[tuple[np.ndarray, np.ndarray, int]] = []
    271 +        while True:
    272 +            temperature = 1.0 if state.moves < temperature_moves else 1e-8
    273 +            mcts = MCTS(
    274 +                model=model,
    275 +                device=device,
    276 +                num_simulations=num_simulations,
    277 +                c_puct=1.5,
    278 +                dirichlet_alpha=0.8,
    279 +                dirichlet_epsilon=0.25,
    280 +            )
    281 +            policy, _ = mcts.search(state, temperature=temperature)
    282 +            if temperature > 0.5:
    283 +                move = int(rng.choice(np.arange(WIDTH), p=policy))
    284 +            else:
    285 +                move = int(np.argmax(policy))
    286 +            history.append((state.to_planes(), policy.astype(np.float32), state.moves % 2))
    287 +            state = state.play(move)
    288 +            terminal = state.terminal_outcome()
    289 +            if terminal is not None:
    290 +                winner = state.winner()
    291 +                lengths.append(state.moves)
    292 +                winners.append(-1 if winner is None else winner)
    293 +                for planes, policy_targets, player_index in history:
    294 +                    if winner is None:
    295 +                        value = 0.0
    296 +                    else:
    297 +                        value = 1.0 if winner == player_index else -1.0
    298 +                    all_planes.append(planes)
    299 +                    all_policies.append(policy_targets)
    300 +                    all_values.append(value)
    301 +                break
    302 +    winner_array = np.array(winners, dtype=np.int32)
    303 +    stats = {
    304 +        "games": float(num_games),
    305 +        "average_length": float(np.mean(lengths)) if lengths else 0.0,
    306 +        "draw_rate": float(np.mean(winner_array == -1)) if len(winner_array) else 0.0,
    307 +        "first_player_win_rate": float(np.mean(winner_array == 0)) if len(winner_array) else 0.0,
    308 +        "second_player_win_rate": float(np.mean(winner_array == 1)) if len(winner_array) else 0.0,
    309 +    }
    310 +    return (
    311 +        np.stack(all_planes).astype(np.float32),
    312 +        np.stack(all_policies).astype(np.float32),
    313 +        np.array(all_values, dtype=np.float32),
    314 +        stats,
    315 +    )

from connect4zero.checkpoint import load_checkpoint
from connect4zero.mcts import MCTS
from connect4zero.oracle import ExactOracle
from connect4zero.state import Connect4State
import torch
checkpoint, model = load_checkpoint('/workspace/runs/connect4_az_oracle_refine/best.pt', 'cuda')
oracle = ExactOracle()
state = Connect4State.initial()
move_history = []
for ply in range(42):
    if state.terminal_outcome() is not None:
        break
    if state.moves % 2 == 0:
        mcts = MCTS(model=model, device='cuda', num_simulations=1024, c_puct=1.5)
        move, policy, value = mcts.select_move(state, temperature=0.0)
        print('model', move + 1, 'value', round(value, 3), flush=True)
    else:
        value, policy, scores = oracle.optimal_policy(state)
        best_score = max(scores.values())
        best_moves = sorted(move for move, score in scores.items() if score == best_score)
        move = best_moves[0]
        print('oracle', move + 1, 'value', value, 'scores', scores, flush=True)
    move_history.append(move)
    state = state.play(move)
    print(state.to_pretty_string(), flush=True)

winner = state.winner()
print('winner', winner)
print('moves', [m + 1 for m in move_history])

    158 +    opening_buckets: dict[int, ArrayTriplet],
