from pathlib import Path
import numpy as np
import torch
import torch.nn.functional as F
from torch.utils.data import DataLoader, TensorDataset
from connect4_az.game import Connect4State, mirror_policy
from connect4_az.solver import ExactSolver
from connect4_az.train import load_checkpoint, resolve_device
solver = ExactSolver()
device = resolve_device()
model, extra = load_checkpoint('logs/20260323_102742/checkpoints/best.pt', device)
model.train()
optimizer = torch.optim.AdamW(model.parameters(), lr=1e-4, weight_decay=1e-5)
lines = [
    '44444432655555323332267623526661117777711',
    '444444322222653335555366635626711111177777',
]
planes = []
policies = []
values = []
seen = set()
for moves in lines:
    for ply in range(len(moves)):
        prefix = moves[:ply]
        if prefix in seen:
            continue
        seen.add(prefix)
        state = Connect4State.from_moves(prefix)
        if state.is_terminal():
            continue
        analysis = solver.analyze(state)
        policy = np.zeros(7, dtype=np.float32)
        policy[list(analysis.best_moves)] = 1.0 / len(analysis.best_moves)
        board = state.encode()
        planes.append(board)
        policies.append(policy)
        values.append(float(analysis.value))
        planes.append(np.ascontiguousarray(board[:, :, ::-1]))
        policies.append(mirror_policy(policy))
        values.append(float(analysis.value))
x = torch.tensor(np.stack(planes), dtype=torch.float32)
y_policy = torch.tensor(np.stack(policies), dtype=torch.float32)
y_value = torch.tensor(np.asarray(values), dtype=torch.float32)
loader = DataLoader(TensorDataset(x, y_policy, y_value), batch_size=64, shuffle=True)
for epoch in range(200):
    for xb, pb, vb in loader:
        xb = xb.to(device)
        pb = pb.to(device)
        vb = vb.to(device)
        optimizer.zero_grad(set_to_none=True)
        logits, value = model(xb)
        policy_loss = -(pb * F.log_softmax(logits, dim=1)).sum(dim=1).mean()
        value_loss = F.mse_loss(value, vb)
        loss = policy_loss + value_loss
        loss.backward()
        optimizer.step()
path = Path('checkpoints/rootline_refined.pt')
path.parent.mkdir(parents=True, exist_ok=True)
checkpoint = torch.load('logs/20260323_102742/checkpoints/best.pt', map_location='cpu')
checkpoint['model_state'] = model.state_dict()
checkpoint['extra'] = dict(checkpoint.get('extra', {}))
checkpoint['extra']['rootline_refined'] = True
torch.save(checkpoint, path)
print(path)
print({'samples': len(x), 'unique_prefixes': len(seen)})
