from bitbully import BitBully, Board
from connect4_az.game import Connect4State
from connect4_az.mcts import MCTSConfig, run_mcts
from connect4_az.training import load_model_from_checkpoint, resolve_device
import numpy as np
def board_from_state(state):
    return Board(state.to_column_array())
def solver_move(agent, state):
    board = board_from_state(state)
    scores = agent.score_all_moves(board)
    best = max(scores.values())
    for move in state.ordered_legal_moves():
        if scores.get(move) == best:
            return move, scores
    raise RuntimeError('no move')
def play_game(checkpoint, agent_player, simulations):
    device = resolve_device('cuda')
    model, _ = load_model_from_checkpoint(checkpoint, device)
    solver = BitBully()
    state = Connect4State()
    history = []
    while not state.is_terminal():
        if state.to_play == agent_player:
            dist, _ = run_mcts(model, state, device,
MCTSConfig(simulations=simulations, dirichlet_epsilon=0.0), add_noise=False)
            move = int(np.argmax(dist))
            history.append(('agent', move + 1, state.to_play))
        else:
            move, scores = solver_move(solver, state)
            history.append(('solver', move + 1, state.to_play))
        state = state.play(move)
    return state.winner(), history
checkpoint = '/home/johnk/Documents/connect4/runs/oracle_fast/best.pt'
for agent_player in (1, -1):
    winner, history = play_game(checkpoint, agent_player=agent_player,
simulations=4096)
    print('agent_player', agent_player, 'winner', winner)
    print('moves', history)
