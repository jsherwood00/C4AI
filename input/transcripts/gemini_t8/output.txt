add: def apply_moves(game, moves_str):
add:     board = game.get_initial_state()
add:     cur_player = 1
add:     
add:     if not moves_str:
add:         return board, cur_player, False, 0
add:         
add:     for i, move_char in enumerate(moves_str):
add:         if not move_char.isdigit():
add:             continue
add:         col = int(move_char) - 1
add:         
add:         # apply move
add:         board = game.get_next_state(board, cur_player, col)
add:         
add:         # Check if this move ended the game
add:         # Since it's from the perspective of the player who just moved
add:         val, term = game.get_value_and_terminated(board, col)
add:         if term:
add:             # If the opponent just moved and the game is terminal
add:             return board, cur_player * -1, True, val
add:             
add:         cur_player *= -1
add:         
add:     return board, cur_player, False, 0
