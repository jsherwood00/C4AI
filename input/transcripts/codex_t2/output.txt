# Find a short sequence where O wins diagonally with negative slope (descending
left-to-right in display).
def has_target(state):
    if not state.terminal or state.winner != -1:
        return False
    board = state.board_matrix()
    # descending in display means row increasing? We'll check both diagonal
directions and require one specific pattern
    for r in range(3,6):
        for c in range(4):
            if all(board[r-i, c+i] == -1 for i in range(4)):
                return True
    return False
seen = set()
queue = deque([(Connect4State.new_game(), [])])
while queue:
